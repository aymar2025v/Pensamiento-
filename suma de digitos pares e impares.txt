
#include <iostream>
using namespace std;
int main()
{

	int ultimoDig,numero,suma1=0,suma2=0;
	cin>>numero;

	while(numero>0) {
		ultimoDig=numero%10;
		numero=numero/10;
		if(ultimoDig%2==0)
			suma1+=ultimoDig;
		else
			suma2+=ultimoDig;
	}
	cout<<"suma de digitos pares: "<<suma1<<'\n';
	cout<<"suma de digitos impares: "<<suma2;

	return 0;
}
